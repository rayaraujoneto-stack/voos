from datetime import date, timedelta
from app import buscar_voos
from notifier import enviar_alerta
from config import DESTINOS, PRECOS_IDEAIS
from database import salvar_voo, ja_notificado, salvar_alerta


def eh_promocao(preco, preco_ideal):
    return preco <= preco_ideal * 0.8


def monitorar():
    hoje = date.today()

    for i in range(1, 10):  # próximos 10 dias
        data = (hoje + timedelta(days=i)).strftime("%Y-%m-%d")

        for destino in DESTINOS:
            try:
                voos = buscar_voos("MAO", destino, data)

                for voo in voos:
                    salvar_voo(destino, data, voo["preco"])

                    preco_ideal = PRECOS_IDEAIS[destino]

                    if eh_promocao(voo["preco"], preco_ideal):

                        if not ja_notificado(destino, data, voo["preco"]):
                            enviar_alerta(voo)
                            salvar_alerta(destino, data, voo["preco"])

            except Exception as e:
                print(f"Erro: {e}")


if __name__ == "__main__":
    monitorar()
